# ====================================
# AI CONFIGURATION (Obrigatório para funcionalidades de IA)
# ====================================

# Provider de IA (openai, openrouter)
# - openai: Usar API oficial da OpenAI
# - openrouter: Usar OpenRouter para acessar múltiplos modelos
AI_PROVIDER=openrouter

# URL base do provider (opcional para OpenAI)
# Para OpenRouter: https://openrouter.ai/api/v1
# Para OpenAI: deixe vazio ou remova esta linha
AI_BASE_URL=https://openrouter.ai/api/v1

# Chave de API do provider
# Para OpenAI: Obtenha em https://platform.openai.com/api-keys
# Para OpenRouter: Obtenha em https://openrouter.ai/keys
OPENAI_API_KEY=sua_chave_aqui

# Modelo a ser usado (OBRIGATÓRIO - não há valor padrão)
# 
# Exemplos para OpenAI:
#   - gpt-4-turbo-preview
#   - gpt-4
#   - gpt-3.5-turbo
#
# Exemplos para OpenRouter:
#   - openai/gpt-4
#   - openai/gpt-3.5-turbo
#   - anthropic/claude-3-opus
#   - anthropic/claude-3-sonnet
#   - google/gemini-pro
#   - meta-llama/llama-3-70b-instruct
#   - openai/gpt-oss-20b:free (modelo gratuito)
OPENAI_MODEL=openai/gpt-3.5-turbo

# Configurações específicas do provider (JSON string, opcional)
# Para OpenRouter, você pode especificar providers preferenciais:
#   {"provider": {"only": ["openai"]}} - Usar apenas OpenAI
#   {"provider": {"only": ["anthropic"]}} - Usar apenas Anthropic
#   {"provider": {"order": ["openai", "anthropic"]}} - Ordem de preferência
# Deixe vazio para usar configuração padrão do provider
AI_PROVIDER_CONFIG=

# Máximo de tokens por requisição (padrão: 25000)
OPENAI_MAX_TOKENS=25000

# Temperatura para geração (0.0-1.0, padrão: 0.7)
# 0.0 = Mais determinístico/consistente
# 1.0 = Mais criativo/variado
OPENAI_TEMPERATURE=0.7

# ====================================
# WEB SCRAPING CONFIGURATION
# ====================================

# Caminho para o WebDriver (opcional - padrão: auto)
# Use 'auto' para detecção automática
WEBDRIVER_PATH=auto

# Timeout para downloads em segundos (padrão: 30)
DOWNLOAD_TIMEOUT=30

# Tempo de espera padrão entre ações em segundos (padrão: 10)
DEFAULT_WAIT_TIME=10

# Número máximo de tentativas em caso de falha (padrão: 3)
MAX_RETRIES=3

# Delay entre tentativas em segundos (padrão: 5)
RETRY_DELAY=5

# ====================================
# FILE PROCESSING CONFIGURATION
# ====================================

# Tamanho máximo de arquivo para upload em bytes (padrão: 10MB)
# 10485760 = 10MB
# 104857600 = 100MB
# 1073741824 = 1GB
FILE_UPLOAD_MAX_SIZE=10485760

# ====================================
# LOGGING CONFIGURATION
# ====================================

# Nível de log: DEBUG, INFO, WARNING, ERROR (padrão: INFO)
LOG_LEVEL=INFO
